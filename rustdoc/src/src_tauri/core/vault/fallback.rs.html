3use crate::core::crypto::Crypto;
4use crate::core::error::AppError;
5use crate::core::logger::Logger;
6use std::fs;
7use std::path::PathBuf;
8use tauri::{AppHandle, Manager};
9
10pub struct FallbackManager;
11
12impl FallbackManager {
13    pub fn get_fallback_path(app: &AppHandle, key: &str) -> Option<PathBuf> {
14        match app.path().app_local_data_dir() {
15            Ok(p) => {
16                let secure_path = p.join(format!("{}.secure", key));
17                Some(secure_path)
18            }
19            Err(e) => {
20                Logger::error(
21                    app,
22                    &format!("[Vault] Failed to resolve data dir: {}", e),
23                    None,
24                );
25                None
26            }
27        }
28    }
29
30    pub fn write_fallback(app: &AppHandle, key: &str, value: &str) -> Result<(), AppError> {
31        if let Some(path) = Self::get_fallback_path(app, key) {
32            let enc_key = super::encryption::EncryptionManager::get_or_create_encryption_key(app)?;
33            let encrypted_value = Crypto::encrypt(&enc_key, value)?;
34
35            if let Err(e) = fs::write(&path, encrypted_value) {
36                Logger::error(
37                    app,
38                    &format!("[Vault] File write failed for {}: {}", key, e),
39                    None,
40                );
41                return Err(AppError::from(e));
42            }
43            Logger::debug(
44                app,
45                &format!("[Vault] Saved {} to disk fallback (encrypted)", key),
46                None,
47            );
48            Ok(())
49        } else {
50            Err(AppError {
51                user_message: "Failed to resolve storage path.".into(),
52                ..Default::default()
53            })
54        }
55    }
56
57    pub fn read_fallback(app: &AppHandle, key: &str) -> Result<String, AppError> {
58        if let Some(path) = Self::get_fallback_path(app, key) {
59            if path.exists() {
60                match fs::read_to_string(&path) {
61                    Ok(encrypted_s) => {
62                        let enc_key =
63                            super::encryption::EncryptionManager::get_or_create_encryption_key(
64                                app,
65                            )?;
66                        return Crypto::decrypt(&enc_key, &encrypted_s);
67                    }
68                    Err(e) => {
69                        Logger::error(
70                            app,
71                            &format!("[Vault] File read failed for {}: {}", key, e),
72                            None,
73                        );
74                        return Err(AppError::from(e));
75                    }
76                }
77            } else {
78                Logger::warn(app, &format!("[Vault] File not found: {:?}", path), None);
79            }
80        }
81        Err(AppError {
82            user_message: "Credential not found in local storage.".into(),
83            error_code: "credentials_missing".into(),
84            ..Default::default()
85        })
86    }
87
88    pub fn delete_fallback(app: &AppHandle, key: &str) -> Result<(), AppError> {
89        if let Some(path) = Self::get_fallback_path(app, key)
90            && path.exists()
91        {
92            fs::remove_file(path).map_err(AppError::from)?;
93        }
94        Ok(())
95    }
96}
