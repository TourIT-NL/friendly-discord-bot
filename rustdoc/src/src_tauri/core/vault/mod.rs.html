26    pub fn get_active_token(app: &AppHandle) -> Result<(String, bool), AppError> {
27        identity::IdentityManager::get_active_token(app)

29
30    #[allow(dead_code)]
31    pub fn get_identity(app: &AppHandle, id: &str) -> Result<DiscordIdentity, AppError> {
32        identity::IdentityManager::get_identity(app, id)
33    }
34
35    pub fn list_identities(app: &AppHandle) -> Vec<DiscordIdentity> {
36        identity::IdentityManager::list_identities(app)
37    }

39    pub fn remove_identity(app: &AppHandle, id: &str) -> Result<(), AppError> {
40        identity::IdentityManager::remove_identity(app, id)
41    }
42
43    pub fn set_credential(app: &AppHandle, key: &str, value: &str) -> Result<(), AppError> {
44        credential::CredentialManager::set_credential(app, key, value)
45    }
46
47    pub fn get_credential(app: &AppHandle, key: &str) -> Result<String, AppError> {
48        credential::CredentialManager::get_credential(app, key)
49    }
50
51    pub fn clear_active_session(app: &AppHandle) -> Result<(), AppError> {
52        let _ = identity::IdentityManager::clear_keyring_entry("active_account");
53        let _ = FallbackManager::delete_fallback(app, "active_account");
54        Logger::info(app, "[Vault] Active session cleared.", None);
55        Ok(())
56    }
57
58    pub fn clear_all_data(app: &AppHandle) -> Result<(), AppError> {
59        Logger::info(app, "[Vault] Initiating full data wipe...", None);
60
61        // Clear all identities
62        let identities = Self::list_identities(app);
63        for identity in identities {
64            let _ = Self::remove_identity(app, &identity.id);
65        }
66
67        // Clear active_account and identity_index
68        let _ = identity::IdentityManager::clear_keyring_entry("active_account");
69        let _ = FallbackManager::delete_fallback(app, "active_account");
70        let _ = identity::IdentityManager::clear_keyring_entry("identity_index");
71        let _ = FallbackManager::delete_fallback(app, "identity_index");
72
73        // Clear credentials
74        let _ = credential::CredentialManager::remove_credential(app, "client_id");
75        let _ = credential::CredentialManager::remove_credential(app, "client_secret");
76
77        Logger::info(app, "[Vault] All application data cleared.", None);
78        Ok(())
79    }
80}
