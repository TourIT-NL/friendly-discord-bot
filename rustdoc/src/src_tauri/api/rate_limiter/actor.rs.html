52    pub async fn run(&mut self) {
53        Logger::info(
54            &self.app_handle,
55            "[LIM] Engine Dispatcher active and resilient",
56            None,
57        );
58
59        // Start background heartbeat task
60        let client_clone = self.client.clone();
61        let app_handle_clone = self.app_handle.clone();
62        tokio::spawn(async move {
63            loop {
64                tokio::time::sleep(Duration::from_secs(45)).await;
65                let op_manager = app_handle_clone.state::<OperationManager>();
66
67                // Only send heartbeat if an operation is currently running or app is in active use
68                if op_manager.state.is_running.load(Ordering::SeqCst) {
69                    let (token, is_bearer) =
70                        match crate::core::vault::Vault::get_active_token(&app_handle_clone) {
71                            Ok(t) => t,
72                            _ => continue,
73                        };
74
75                    let mut req = client_clone
76                        .post("https://discord.com/api/v9/users/@me/meaningfully-online");
77                    if is_bearer {
78                        req = req.header(header::AUTHORIZATION, format!("Bearer {}", token));
79                    } else {
80                        req = req.header(header::AUTHORIZATION, token);
81                    }
82
83                    let _ = req.send().await;
84                }
85            }
86        });
87
88        while let Some(request) = self.inbox.recv().await {
89            // Block telemetry calls entirely at the dispatcher level
90            if request.url.contains("/beaker") || request.url.contains("/metrics") {
91                let _ = request.response_tx.send(Err(AppError {
92                    user_message: "Telemetry blocked by Privacy Guard".into(),
93                    error_code: "telemetry_blocked".into(),
94                    ..Default::default()
95                }));
96                continue;
97            }
98
99            let client = self.client.clone();
100            let buckets_map = self.buckets.clone();
101            let global_throttle = self.global_reset_at.clone();
102            let app_handle = self.app_handle.clone();
103            let route = Self::get_route(&request.url);
104
105            tokio::spawn(async move {
106                let mut retry_count = 0;
107                const MAX_RETRIES: u32 = 3;
108
109                loop {
110                    // Apply human-like jitter before every request
111                    let jitter = rand::thread_rng().gen_range(50..250);
112                    tokio::time::sleep(Duration::from_millis(jitter)).await;
113
114                    let now = Instant::now();
115
116                    // 1. Global Wait
117                    {
118                        let global = global_throttle.lock().await;
119                        if now < *global {
120                            let wait = *global - now;
121                            tokio::time::sleep(wait).await;
122                            continue;
123                        }
124                    }
125
126                    // 2. Bucket Synchronization
127                    let bucket_arc = {
128                        let mut map = buckets_map.lock().await;
129                        map.entry(route.clone())
130                            .or_insert_with(|| Arc::new(Mutex::new(BucketInfo::default())))
131                            .clone()
132                    };
133
134                    {
135                        let mut bucket = bucket_arc.lock().await;
136                        if now >= bucket.reset_at {
137                            bucket.remaining = bucket.limit;
138                        }
139
140                        if bucket.remaining == 0 {
141                            let wait = bucket.reset_at.saturating_duration_since(now);
142                            if !wait.is_zero() {
143                                drop(bucket);
144                                tokio::time::sleep(wait + Duration::from_millis(100)).await;
145                                continue;
146                            }
147                        }
148                        bucket.remaining = bucket.remaining.saturating_sub(1);
149                    }
150
151                    // 3. Execution
152                    let mut req_builder = client.request(request.method.clone(), &request.url);
153
154                    // Contextual Referer Injection
155                    if request.url.contains("/messages") {
156                        req_builder =
157                            req_builder.header("referer", "https://discord.com/channels/@me");
158                    } else if request.url.contains("/settings") || request.url.contains("/harvest")
159                    {
160                        req_builder =
161                            req_builder.header("referer", "https://discord.com/settings/privacy");
162                    } else if request.url.contains("/billing") {
163                        req_builder =
164                            req_builder.header("referer", "https://discord.com/settings/billing");
165                    }
166
167                    if request.is_bearer {
168                        req_builder = req_builder.header(
169                            header::AUTHORIZATION,
170                            format!("Bearer {}", request.auth_token),
171                        );
172                    } else {
173                        req_builder =
174                            req_builder.header(header::AUTHORIZATION, &request.auth_token);
175                    }
176
177                    if let Some(body) = request.body.clone() {
178                        req_builder = req_builder.json(&body);
179                    }
180
181                    match req_builder.send().await {
182                        Ok(response) => {
183                            let status = response.status();
184                            let is_429 = status.as_u16() == 429;
185
186                            Self::process_headers(
187                                &app_handle,
188                                &route,
189                                &response,
190                                &bucket_arc,
191                                &global_throttle,
192                                is_429,
193                            )
194                            .await;
195
196                            if is_429 {
197                                Logger::warn(
198                                    &app_handle,
199                                    &format!("[LIM] 429 received for {}", route),
200                                    None,
201                                );
202                                continue; // Retry after rate limit
203                            }
204
205                            if !status.is_success()
206                                && status.is_server_error()
207                                && retry_count < MAX_RETRIES
208                            {
209                                retry_count += 1;
210                                tokio::time::sleep(Duration::from_secs(retry_count as u64)).await;
211                                continue; // Retry after server error
212                            }
213
214                            // Deserialize response body within the actor
215                            let result = if status.is_success() {
216                                if status == reqwest::StatusCode::NO_CONTENT {
217                                    Ok(serde_json::json!({}))
218                                } else {
219                                    response
220                                        .json::<serde_json::Value>()
221                                        .await
222                                        .map_err(AppError::from)
223                                }
224                            } else {
225                                // For non-success responses, try to read body as JSON for error details
226                                // Otherwise, create a generic AppError
227                                response
228                                    .json::<serde_json::Value>()
229                                    .await
230                                    .map_err(AppError::from)
231                                    .and_then(|json_body| {
232                                        Err(AppError {
233                                            user_message: format!("API error: {}", json_body),
234                                            error_code: format!("api_http_{}", status.as_u16()),
235                                            technical_details: Some(json_body.to_string()),
236                                        })
237                                    })
238                                    .map_err(|_| AppError {
239                                        user_message: format!("API error with status {}", status),
240                                        error_code: format!("api_http_{}", status.as_u16()),
241                                        technical_details: Some(format!(
242                                            "Response status: {}",
243                                            status
244                                        )),
245                                    })
246                            };
247
248                            let _ = request.response_tx.send(result);
249                            break;
250                        }
251                        Err(e) => {
252                            if retry_count < MAX_RETRIES {
253                                retry_count += 1;
254                                tokio::time::sleep(Duration::from_secs(retry_count as u64)).await;
255                                continue;
256                            }
257                            let _ = request.response_tx.send(Err(AppError::from(e)));
258                            break;
259                        }
260                    }
261                }
262            });
263        }
264    }

266    async fn process_headers(
267        app: &tauri::AppHandle,
268        route: &str,
269        response: &Response,
270        bucket_arc: &Arc<Mutex<BucketInfo>>,
271        global_throttle: &Arc<Mutex<Instant>>,
272        is_429: bool,
273    ) {
274        let headers = response.headers();
275        let now = Instant::now();
276        let mut bucket = bucket_arc.lock().await;
277
278        if is_429 {
279            bucket.consecutive_429s += 1;
280        } else {
281            bucket.consecutive_429s = 0;
282        }
283
284        if let Some(rem) = headers
285            .get("X-RateLimit-Remaining")
286            .and_then(|h| h.to_str().ok())
287            .and_then(|s| s.parse::<u32>().ok())
288        {
289            bucket.remaining = rem;
290        }
291        if let Some(reset) = headers
292            .get("X-RateLimit-Reset-After")
293            .and_then(|h| h.to_str().ok())
294            .and_then(|s| s.parse::<f32>().ok())
295        {
296            bucket.reset_at = now + Duration::from_secs_f32(reset);
297        }
298        if let Some(lim) = headers
299            .get("X-RateLimit-Limit")
300            .and_then(|h| h.to_str().ok())
301            .and_then(|s| s.parse::<u32>().ok())
302        {
303            bucket.limit = lim;
304        }
305
306        if is_429 {
307            let retry_after = headers
308                .get("Retry-After")
309                .and_then(|h| h.to_str().ok())
310                .and_then(|s| s.parse::<f32>().ok())
311                .unwrap_or(1.0);
312
313            let mut wait = Duration::from_secs_f32(retry_after);
314            if bucket.consecutive_429s > 1 {
315                wait += Duration::from_secs(3u64.pow(bucket.consecutive_429s.min(5)));
316            }
317
318            if headers
319                .get("X-RateLimit-Global")
320                .and_then(|h| h.to_str().ok())
321                == Some("true")
322            {
323                let mut g = global_throttle.lock().await;
324                *g = now + wait;
325                Logger::error(
326                    app,
327                    &format!("[LIM] GLOBAL RATE LIMIT. Locking for {:?}", wait),
328                    None,
329                );
330            } else {
331                bucket.remaining = 0;
332                bucket.reset_at = now + wait;
333                Logger::warn(
334                    app,
335                    &format!("[LIM] Route '{}' limited for {:?}", route, wait),
336                    None,
337                );
338            }
339        }
340    }
