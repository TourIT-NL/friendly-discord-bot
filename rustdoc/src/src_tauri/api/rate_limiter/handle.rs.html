3use crate::api::rate_limiter::types::ApiRequest;
4use crate::core::error::AppError;
5use reqwest::Method;
6use tokio::sync::{mpsc, oneshot};
7
8#[derive(Clone)]
9pub struct ApiHandle {
10    pub(crate) tx: mpsc::Sender<ApiRequest>,
11}
12
13impl ApiHandle {
14    pub fn new(tx: mpsc::Sender<ApiRequest>) -> Self {
15        Self { tx }
16    }
17
18    pub async fn send_request(
19        &self,
20        method: Method,
21        url: &str,
22        body: Option<serde_json::Value>,
23        auth_token: &str,
24        is_bearer: bool,
25    ) -> Result<serde_json::Value, AppError> {
26        let (response_tx, response_rx) = oneshot::channel();
27
28        let api_request = ApiRequest {
29            method,
30            url: url.to_string(),
31            body,
32            auth_token: auth_token.to_string(),
33            is_bearer,
34            response_tx,
35        };
36
37        self.tx.send(api_request).await.map_err(|_| AppError {
38            user_message: "Rate limiter connection failure.".to_string(),
39            error_code: "limiter_offline".to_string(),
40            technical_details: None,
41        })?;
42
43        response_rx.await.map_err(|_| AppError {
44            user_message: "Rate limiter communication timeout.".to_string(),
45            error_code: "limiter_timeout".to_string(),
46            technical_details: None,
47        })?
48    }
49}
